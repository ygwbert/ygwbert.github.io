迭代法勾股数

#include <iostream>
#include <math.h>
using namespace std;

int main() {
	int a, b;
	cout << "请输入左边界";
	cin >> a;
	cout << "请输入右边界";
	cin >> b;
	if (a >= b) {
		cout << "边界错误输入" << endl;
		exit(1);
	}
	for (int i = a; i <= b-2; i++) {  //设置为b-2，因为除了3这个和5相差二的勾股数，再没有其他的只比斜边小二的数了		
		for (int j = i+1; j <= b-1; j++) {     //同上，除了4，没有只比斜边小于一的勾股数，避免重复，所以设置j等于i+1
			int temp = i * i + j * j;
			
			int k = sqrt(i*i + j * j); 

			if (k >= a && k <= b&&k*k==temp) {
				cout << i << '\0' << j << '\0' << k << endl;
			}
		}
	}
}

各种方法遍历数组

#include <iostream>
using namespace std;
int main() {
	int a[3][4] = { {1,1,1,6},{2,2,2,7},{3,3,3,8} };
	int *p = nullptr;
	p = &a[0][0];                      //指针变量遍历二维数组元素
	for (; p < a[0] + 12; p++) {
		cout << *p << '\0';
	}
	cout << endl;
	int(*t)[4];
	t = a;
	for (; t < a + 3; t++) {
		int *n = *t;
		for (; n < *t + 4; n++) {
			cout << *n <<'\0';
		}
	}

	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 4; j++) {
			cout << *(*(a + i) + j) << '\0';
		}
}}
