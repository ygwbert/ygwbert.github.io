函数指针和指针函数

#include <iostream>
using namespace std;

int *func();
int main() {
	int *p;
	p = func();
	cout << "这是被调用函数中定义的变量的地址" << p << endl;
}

int *func() {
	int a = 1, *p;
	cout << "这是一个返回指针变量的函数" << endl;
	p = &a;
	return p;
}
#include <iostream>
using namespace std;

int print();
int main() {
	int(*p)();
	p = print;//此时函数名可以和p互换，函数名即函数第一条指令所在的地址
	p();
}

int print() {
	cout << "打印一行字" << endl;
	return 0;
}
